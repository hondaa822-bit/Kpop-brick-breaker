<script>

const canvas = document.getElementById("game");
const ctx = canvas.getContext("2d");

let paddle = {x:400,y:520,w:140,h:15,speed:8};
let ball = {x:450,y:500,r:8,dx:4,dy:-4};

let bricks=[];
let particles=[];
let running=false;

const colors=[
"#ff4fd8",
"#5ef2ff",
"#ffe66d",
"#8d7bff",
"#7cff95"
];

function createBricks(){

bricks=[];

for(let r=0;r<6;r++){

for(let c=0;c<10;c++){

bricks.push({

x:70+c*75,
y:70+r*35,
w:65,
h:20,
color:colors[(r+c)%colors.length]

});

}

}

}

function drawPaddle(){

ctx.shadowColor="#5ef2ff";
ctx.shadowBlur=20;

let grad=ctx.createLinearGradient(
paddle.x,
paddle.y,
paddle.x+paddle.w,
paddle.y
);

grad.addColorStop(0,"#ff4fd8");
grad.addColorStop(1,"#5ef2ff");

ctx.fillStyle=grad;

ctx.fillRect(paddle.x,paddle.y,paddle.w,paddle.h);

ctx.shadowBlur=0;

}

function drawBall(){

ctx.shadowColor="#ffe66d";
ctx.shadowBlur=20;

ctx.beginPath();
ctx.arc(ball.x,ball.y,ball.r,0,Math.PI*2);
ctx.fillStyle="#ffe66d";
ctx.fill();

ctx.shadowBlur=0;

}

function drawBricks(){

for(let b of bricks){

ctx.shadowColor=b.color;
ctx.shadowBlur=10;

ctx.fillStyle=b.color;
ctx.fillRect(b.x,b.y,b.w,b.h);

ctx.shadowBlur=0;

ctx.fillStyle="white";
ctx.font="12px Arial";
ctx.fillText("♪",b.x+28,b.y+14);

}

}

function createParticles(x,y,color){

for(let i=0;i<12;i++){

particles.push({

x:x,
y:y,
dx:(Math.random()-0.5)*5,
dy:(Math.random()-0.5)*5,
life:30,
color:color

});

}

}

function drawParticles(){

for(let p of particles){

ctx.fillStyle=p.color;
ctx.fillRect(p.x,p.y,3,3);

}

}

function updateParticles(){

for(let i=particles.length-1;i>=0;i--){

let p=particles[i];

p.x+=p.dx;
p.y+=p.dy;
p.life--;

if(p.life<=0) particles.splice(i,1);

}

}

function collision(){

for(let i=bricks.length-1;i>=0;i--){

let b=bricks[i];

if(

ball.x>b.x &&
ball.x<b.x+b.w &&
ball.y>b.y &&
ball.y<b.y+b.h

){

ball.dy*=-1;

createParticles(
b.x+b.w/2,
b.y+b.h/2,
b.color
);

bricks.splice(i,1);

}

}

}

function update(){

ball.x+=ball.dx;
ball.y+=ball.dy;

if(ball.x<ball.r || ball.x>canvas.width-ball.r)
ball.dx*=-1;

if(ball.y<ball.r)
ball.dy*=-1;

if(ball.y>canvas.height){

resetBall();

}

if(

ball.x>paddle.x &&
ball.x<paddle.x+paddle.w &&
ball.y>paddle.y

){

ball.dy=-Math.abs(ball.dy);

}

collision();

updateParticles();

}

function resetBall(){

ball.x=450;
ball.y=500;
ball.dx=4;
ball.dy=-4;

}

function draw(){

ctx.clearRect(0,0,canvas.width,canvas.height);

drawBricks();
drawPaddle();
drawBall();
drawParticles();

}

function loop(){

if(!running) return;

update();
draw();

requestAnimationFrame(loop);

}

function startGame(){

document.getElementById("start").style.display="none";

running=true;

createBricks();

loop();

}

document.addEventListener("mousemove",e=>{

let rect=canvas.getBoundingClientRect();

paddle.x=(e.clientX-rect.left)-paddle.w/2;

});

</script>
